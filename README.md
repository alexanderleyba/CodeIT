# Setup:
1. Import database dump db.sql
2. go Core/init.php
3. in GLOBALS array change host/username/password/db values to your own.

    $GLOBALS['config'] = array(
    	'mysql' => array(
    		'host' => '127.0.0.1',
    		'username' => 'root',
    		'password' => '',
    		'db' => 'test'
    	)
    );
4. Done!
